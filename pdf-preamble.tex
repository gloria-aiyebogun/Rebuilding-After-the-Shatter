% ============================================================
% PDF Preamble — Rebuilding after the Shatter
% Professional self-help book formatting
% Gloria Aiyebogun
% ============================================================

% --- Paragraph spacing (nonfiction style: space between paragraphs, no indent) ---
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.75em}

% --- Chapter & section heading styles ---
\usepackage{titlesec}

% --- Running headers / footers ---
\usepackage{fancyhdr}

% --- Accent colour for decorative rules ---
\usepackage{xcolor}
\definecolor{warmgold}{RGB}{155, 126, 90}

% ============================================================
% CHAPTER TITLE FORMAT
% Centered, large bold text with a short decorative rule below
% ============================================================
\titleformat{\chapter}[block]
  {\normalfont\Large\bfseries\centering}
  {}
  {0pt}
  {}
  [\vspace{0.5em}%
   {\color{warmgold}\rule{0.3\textwidth}{0.6pt}}%
   \vspace{1.5em}]

% Vertical space around chapter titles
\titlespacing*{\chapter}{0pt}{*1}{*4}

% ============================================================
% SECTION HEADING FORMAT  (## in markdown)
% Bold italic — warm, personal feel for self-help sections
% ============================================================
\titleformat{\section}
  {\normalfont\large\bfseries\itshape}
  {}
  {0pt}
  {}

\titlespacing*{\section}{0pt}{*2}{*1}

% ============================================================
% SUBSECTION HEADING FORMAT  (### in markdown)
% Bold, slightly smaller — used for reflection exercises, etc.
% ============================================================
\titleformat{\subsection}
  {\normalfont\normalsize\bfseries}
  {}
  {0pt}
  {}

\titlespacing*{\subsection}{0pt}{*1.5}{*0.5}

% ============================================================
% RUNNING HEADERS & FOOTERS
% Left pages: book title  |  Right pages: chapter title
% Page numbers at outer bottom edge
% ============================================================
\pagestyle{fancy}
\fancyhf{}

% Even (left) pages: book title on left
\fancyhead[LE]{\small\itshape Rebuilding after the Shatter}
% Odd (right) pages: chapter name on right
\fancyhead[RO]{\small\itshape\nouppercase{\leftmark}}
% Page numbers — outer bottom
\fancyfoot[LE,RO]{\small\thepage}

\renewcommand{\headrulewidth}{0.3pt}
\renewcommand{\footrulewidth}{0pt}

% Chapter opening pages use 'plain' style in LaTeX book class.
% Override to show only the page number (no header line).
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[LE,RO]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% ============================================================
% BLOCKQUOTE STYLING
% Reflection exercises appear as blockquotes.
% Add a warm left border inside the PDF.
% ============================================================
\usepackage{mdframed}
\usepackage{environ}

% Redefine quote environment used by pandoc for blockquotes
\renewenvironment{quote}{%
  \begin{mdframed}[
    leftline=true,
    rightline=false,
    topline=false,
    bottomline=false,
    linecolor=warmgold,
    linewidth=2pt,
    innerleftmargin=12pt,
    innerrightmargin=6pt,
    innertopmargin=6pt,
    innerbottommargin=6pt,
    backgroundcolor=white
  ]
  \itshape
}{%
  \end{mdframed}%
}
